# Manager Agent container — pinned for reproducible builds.
# This list is consumed by the Manager Agent image (built from the
# agents/ tree), NOT the root-level `requirements.txt` (which is for
# the MCP core container + host-side CLI install).

# Google ADK 2.0 with A2A + extensions — the Manager Agent runtime.
google-adk[a2a,extensions]==2.0.0

# GCS — used for shared workspace artifact upload by some skills.
google-cloud-storage==2.18.2

# Multipart upload support (file attachments in chat).
python-multipart==0.0.27

# Model Context Protocol — manager agent consumes MCP tools.
mcp==1.9.4

# LiteLLM — pluggable non-Google LLM backend.
# Must stay within google-adk[extensions]==2.0.0's range (>=1.83.7,<=1.83.14).
litellm==1.83.14

# Config + HTTP + DB drivers (loaded lazily by ADK / chatbot SDK based
# on ADK_SESSION_SERVICE_URI scheme).
python-dotenv==1.2.2
# httpx 0.28.1: required exactly by litellm 1.83.14 (==0.28.1) and accepted by
# google-adk 2.0.0 (<1,>=0.27) + mcp 1.9.4 (>=0.27). Pinning 0.27.2 collided
# with litellm and made the Manager image unbuildable (pip ResolutionImpossible).
httpx==0.28.1
aiosqlite==0.21.0
asyncpg==0.30.0
