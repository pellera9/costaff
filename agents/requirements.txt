# Manager Agent container — pinned for reproducible builds.
# This list is consumed by the Manager Agent image (built from the
# agents/ tree), NOT the root-level `requirements.txt` (which is for
# the MCP core container + host-side CLI install).
#
# litellm / mcp / httpx below are TRANSITIVE deps of google-adk[a2a,extensions].
# We pin them explicitly anyway: leaving them to float let the build drift over
# time (prod-test's cached image froze a working set, so a fresh rebuild on a
# clean index broke on too-old/incompatible versions). Keep them pinned to the
# verified-good set below. Verified on a fresh GCE build 2026-05-30 (E2E pass).

# Google ADK 2.1.0 with A2A + extensions — the Manager Agent runtime.
google-adk[a2a,extensions]==2.1.0

# GCS — shared workspace artifact upload by some skills.
google-cloud-storage==2.18.2

# Multipart upload support (file attachments in chat).
python-multipart==0.0.27

# Model Context Protocol — manager agent consumes MCP tools.
# google-adk[extensions] requires mcp>=1.16.0; its mcp_tool/__init__.py imports
# StreamableHTTPConnectionParams etc. inside a try/except, so anything below
# 1.16.0 (e.g. 1.9.4) makes that import fail silently → McpToolset is never
# re-exported → the Manager fails to load. 1.27.1 = the version prod-test runs.
mcp==1.27.1

# LiteLLM — pluggable non-Google LLM backend.
# google-adk[extensions]==2.1.0 requires litellm>=1.83.7,<=1.83.14; pin the top.
litellm==1.83.14

# HTTP + DB drivers.
python-dotenv==1.2.2
# litellm 1.83.14 requires httpx==0.28.1 exactly; google-adk (<1,>=0.27) and
# mcp (>=0.27) both accept it. An older pin (0.27.2) made the image unbuildable
# (pip ResolutionImpossible).
httpx==0.28.1
# Session-service drivers for ADK_SESSION_SERVICE_URI (sqlite / postgres).
aiosqlite==0.21.0
asyncpg==0.30.0
